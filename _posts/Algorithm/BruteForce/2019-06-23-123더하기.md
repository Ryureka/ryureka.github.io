	public static int go(int index,int sum,int N) {

#### 불필요한 변수제거 및 cnt를 전역변수로 처리하여 계산
{% highlight java %}
import java.util.Scanner;

public class Main {
	static int cnt;
	public static void main(String[] args) {
		Scanner sc=new Scanner(System.in);
		int T=sc.nextInt();
		for (int t = 0; t < T; t++) {
			cnt=0;
			int n=sc.nextInt();
			go(n,0);
			System.out.println(cnt);
		}		
	}

	public static void go(int n,int sum) {
		if(sum>n) {
			return;
		}
		else if(sum==n) {
			cnt++;
		}
		for (int i = 1; i <= 3; i++) {
			go(n,sum+i);			
		}
	}

}
{% endhighlight %}
